function joinCommunity() {
  const dev = new Developer({
    username: 'ваш_никнейм',
    email: 'ваш@email.com',
    skills: ['JavaScript', 'Python', 'React'],
    isAwesome: true
  });

  dev.register();
  dev.connect();
  dev.create();
}
